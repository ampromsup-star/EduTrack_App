const SS_ID = SpreadsheetApp.getActiveSpreadsheet().getId(); function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; let result; if (action === 'recordAttendance') { result = recordAttendance(data.studentId, data.courseId); } else if (action === 'login') { result = checkLogin(data.user, data.pass); } else { result = { status: "error", message: "ไม่พบ Action ที่ระบุ" }; } return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function recordAttendance(studentId, courseId) { const ss = SpreadsheetApp.openById(SS_ID); const sheet = ss.getSheetByName('Attendance'); const now = new Date(); const dateStr = Utilities.formatDate(now, "GMT+7", "yyyy-MM-dd"); const timeStr = Utilities.formatDate(now, "GMT+7", "HH:mm"); let status = (timeStr > "08:15") ? "สาย" : "ตรง"; sheet.appendRow([sheet.getLastRow(), studentId, courseId, dateStr, timeStr, status]); return { status: "success", info: status, time: timeStr }; } function checkLogin(user, pass) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName('Users'); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][4] == user && data[i][5] == pass) { return { status: "success", user: { name: data[i][1], role: data[i][2] } }; } } return { status: "error", message: "รหัสผ่านผิด" }; } // เพิ่มต่อท้ายใน Code.gs เดิม function doGet(e) { return HtmlService.createTemplateFromFile('index').evaluate() .addMetaTag('viewport', 'width=device-width, initial-scale=1'); } // ฟังก์ชันดึงรายการงานทั้งหมด function getAssignments() { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName('Assignments'); const data = sheet.getDataRange().getValues(); const headers = data.shift(); return data.map((row, index) => ({ row: index + 2, // เก็บตำแหน่งแถวไว้สำหรับอ้างอิงตอนลบ/แก้ไข id: row[0], courseId: row[1], title: row[2], fullScore: row[3], passScore: row[4], dueDate: row[5] ? Utilities.formatDate(new Date(row[5]), "GMT+7", "yyyy-MM-dd") : "" })); } // ฟังก์ชันลบงาน function deleteAssignment(row) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName('Assignments'); sheet.deleteRow(row); return { status: "success" }; } // ฟังก์ชันเพิ่มงานใหม่ function addAssignment(data) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName('Assignments'); sheet.appendRow([ "ASGN-" + Utilities.getUuid().substring(0,5), // สุ่ม ID data.courseId, data.title, data.fullScore, data.passScore, data.dueDate ]); return { status: "success" }; } // อัปเดต doPost ให้รองรับ Action ใหม่ๆ function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; if (action === 'getAssignments') return jsonResponse(getAssignments()); if (action === 'deleteAssignment') return jsonResponse(deleteAssignment(data.row)); if (action === 'addAssignment') return jsonResponse(addAssignment(data)); if (action === 'recordAttendance') return jsonResponse(recordAttendance(data.studentId, data.courseId)); if (action === 'login') return jsonResponse(checkLogin(data.user, data.pass)); } catch (err) { return jsonResponse({status: "error", message: err.toString()}); } } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
